  <li class="nav-item">
    <a class="nav-link active" href="#home" data-toggle="tab" role="tab">
      <i class="fe-icon-home"></i>
      &nbsp;Home
    </a>
  </li>
  <li class="nav-item">
    <a class="nav-link" href="#profile" data-toggle="tab" role="tab">
      <i class="fe-icon-user"></i>
      &nbsp;Profile
    </a>
  </li>
  <li class="nav-item dropdown">
    <a class="nav-link dropdown-toggle" href="#" data-toggle="dropdown" role="button">Dropdown</a>
    <div class="dropdown-menu">
      <a class="dropdown-item" href="#messages" data-toggle="tab" role="tab">Messages</a>
      <a class="dropdown-item" href="#settings" data-toggle="tab" role="tab">Settings</a>
    </div>
  </li>
</ul>

      i.fe-icon-home
      | &nbsp;Home
  li.nav-item
    a(href="#profile" class="nav-link" data-toggle="tab" role="tab")
