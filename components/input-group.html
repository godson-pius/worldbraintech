<!-- Multiple Button Addons -->
<div class="input-group">
  <input class="form-control border-right-0" type="text" placeholder="Buttons on the right">
  <div class="input-group-append">
    <button class="btn btn-secondary" type="button"><i class="fe-icon-edit text-lg"></i></button>
    <button class="btn btn-success" type="button"><i class="fe-icon-check-circle text-lg"></i></button>
    <button class="btn btn-danger" type="button"><i class="fe-icon-trash-2 text-lg"></i></button>
  </div>
</div>

<!-- Dropdown Addon on the Right -->
<div class="input-group">
  <input class="form-control" type="text" placeholder="Dropdown on the right">
  <div class="input-group-append">
    <button class="btn btn-primary dropdown-toggle" type="button" data-toggle="dropdown">Dropdown</button>
    <nav class="dropdown-menu">
      <a class="dropdown-item" href="#">Action</a>
      <a class="dropdown-item" href="#">Another action</a>
      <a class="dropdown-item" href="#">Something else here</a>
    </nav>
  </div>
</div>
//- Button Addon on the Left
.input-group
  .input-group-prepend
    button(type="button").btn.btn-primary Button
  input(type="text" placeholder="Button on the left").form-control
.input-group.form-group
  input(type="text" placeholder="Button on the right").form-control
  .input-group-append
    button(type="button").btn.btn-primary Button

//- Multiple Button Addons
